function doDispatchScan() { if (!currentDispatchClient) { toast('Select a request first','err'); return; } var serial = document.getElementById('dispatch-serial').value.trim(); if (!serial) { toast('Enter Serial No','err'); return; } var resEl = document.getElementById('dispatch-result'); if (dispatchedItems.find(function(i){return i.serial.toUpperCase()===serial.toUpperCase();})) { toast('Already added: '+serial,'warn'); document.getElementById('dispatch-serial').value=''; return; } callGAS('scanDispatch', {serial: serial, clientName: currentDispatchClient}, function(res) { if (res.ok && res.found) { // DESCRIPTION CHECK: must match a requested item if (currentRequestLineItems && currentRequestLineItems.length > 0) { function norm(s){ return s.toLowerCase().replace(/\s+/g,'').replace(/[-_]/g,''); } var resN = norm(res.description); var matchedLine = null; currentRequestLineItems.forEach(function(l) { var lN = norm(l.description); if (resN.indexOf(lN.substring(0,10)) >= 0 || lN.indexOf(resN.substring(0,10)) >= 0) matchedLine = l; }); if (!matchedLine) { resEl.innerHTML = '
'+ 'WRONG ITEM
Scanned: '+esc(res.description)+'
'+ 'Requested: '+esc(currentRequestLineItems.map(function(l){return l.description;}).join(' / '))+ '
'; document.getElementById('dispatch-serial').value=''; return; } // QUANTITY CHECK function norm2(s){ return s.toLowerCase().replace(/\s+/g,'').replace(/[-_]/g,''); } var lN2 = norm2(matchedLine.description); var alreadyCount = dispatchedItems.filter(function(it){ var itN = norm2(it.description); return itN.indexOf(lN2.substring(0,10)) >= 0 || lN2.indexOf(itN.substring(0,10)) >= 0; }).length; if (alreadyCount >= parseInt(matchedLine.quantity)) { resEl.innerHTML = '
'+ 'QUANTITY EXCEEDED
Already scanned '+alreadyCount+'/'+matchedLine.quantity+' of '+esc(matchedLine.description)+ '
'; document.getElementById('dispatch-serial').value=''; return; } } // FIFO WARNING if (res.fifoWarning) { window._pendingItem = {rowIndex:res.rowIndex,stockSheet:res.stockSheet,serial:res.serial,partNo:res.partNo,description:res.description,location:res.location}; resEl.innerHTML = '
'+ 'FIFO WARNING
Older item of this type exists. Scan that one first.
Location of this item: '+esc(res.location)+ '
'+ ''+ ''; document.getElementById('dispatch-serial').value=''; return; } addDispatchItem({rowIndex:res.rowIndex,stockSheet:res.stockSheet,serial:res.serial,partNo:res.partNo,description:res.description,location:res.location},resEl); } else if (res.notFound) { resEl.innerHTML='
Not found: '+esc(serial)+'
'+ ''+ ''; } else { resEl.innerHTML='
'+esc(res.msg)+'
'; } }); } function addPendingDispatchItem() { if (window._pendingItem) { addDispatchItem(window._pendingItem, document.getElementById('dispatch-result')); window._pendingItem=null; } } function addDispatchItem(item, resEl) { dispatchedItems.push(item); renderDispatchedList(); resEl.innerHTML='
Added: '+esc(item.description)+' -- Location: '+esc(item.location)+'
'; setTimeout(function(){resEl.innerHTML='';},3000); autoSave('dispatch', dispatchedItems, currentDispatchClient); }
